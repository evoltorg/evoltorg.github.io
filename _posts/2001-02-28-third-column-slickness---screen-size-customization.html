\nThe techniques involves using a ILayer or IFrame (NS4 and others respectively) to bring in another page of content. One side effect is that this is also a technique for including content from a separate site. \n

\n

\nIn this case, I'm using table with alignment wrapped around the iframe/ilayer to avoid having to alter any of the other markup on the page\n

\n
if (document.layers    document.getElementById) {\n     screenwidth = window.innerWidth;\n     screenheight  = window.innerHeight;\n}\nif (document.all ) {\n     screenwidth = document.body.clientWidth;\n     screenheight = document.body.clientHeight;\n}
if(screenwidth > 800) {\n	if(document.layers) {\n		document.write('<table align="right" width="180"><tr><td>  </td><td><ilayer src="/blogger/featured.cfm" width="180" height="' + screenheight + '" scrolling="no"></ilayer></td></tr></table>');\n	} else {\n		if(document.all    document.getElementById) {\n			document.write('<table align="right" width="180"><tr><td>  </td><td><iframe src="/blogger/featured.cfm" width="180" height="' + screenheight + '" scrolling="no"></iframe></td></tr></table>');\n		}\n	}	\n}\n
