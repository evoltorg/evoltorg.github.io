You can easy set callbacks, bind to any interface!

See:

$curl->setopt(CURLOPT_HTTPHEADER, \@headers_arr);

$curl->setopt(CURLOPT_FOLLOWLOCATION, 1);

$curl->setopt(CURLOPT_MAXREDIRS, 4);

$curl->setopt(CURLOPT_HTTPGET, 1);

$curl->setopt(CURLOPT_URL, $url);

$curl->setopt(CURLOPT_TIMEOUT, $http_timeout);

$curl->setopt(CURLOPT_INTERFACE, $ip) if $ip;

if ($proxy && $proxy =~ /^.*:\d+$/)

{

$curl->setopt(CURLOPT_PROXY, "http://$proxy") if $ip;

}

$curl->setopt(CURLOPT_COOKIE, $cookie) if $cookie;
